## 一、简单的理解响应式

拦截

```js
 const data = { name: "余光" };
            const objP = new Proxy(data, {
                get(target, key) {
                    return target[key];
                },
                set(target, key, newVal) {
                    target[key] = newVal;
                    callback();
                    return target[key];
                },
            });
            function getData() {
                console.log("get-name:", objP.name);
            }
            function setData() {
                objP.name += "1";
            }
            function callback() {
                console.log("触发拦截:");
            }
```

响应

```html
<button id="btn" onclick="addData()"></button>

<script>
    const data = { index: 1 };

    }
    // 注册响应函数
    effect();
```

### 响应函数与被响应数据应该建立准确的关系


1. 代理改造为通用方式
2. 将响应关系 对象 -> map -> key: effects[]
    * WeakMap -> Object : Map
    * Map -> key : Set

```html
<span id="wrapper"></span>
        <button id="btn" onclick="addData()">+1</button>
        <button id="btn" onclick="curData()">-1</button>
        <div></div>
        <script>
            const data = { index: 1 };
            // “桶”改造
            const bucket = new WeakMap();

            // 改造1 - 响应函数
            let activeEffect;
            function effect(fn) {
                activeEffect = fn;
                fn();
            }

            // 改造1 - 拦截代理
            const objP = new Proxy(data, {
                get(target, key) {
                    // 没有方法直接返回
                    if (!activeEffect) return;

                    // 获取响应对象内部的响应关系
                    let dependMap = bucket.get(target);
                    if (!dependMap) {
                        bucket.set(target, new Map());
                        dependMap = bucket.get(target);
                    }
                    // 获取对应属性的响应函数列表
                    // 存储着与key有关联的响应函数
                    let dependList = dependMap[key];
                    if (!dependList) {
                        dependMap.set(key, new Set());
                        dependList = dependMap.get(key);
                    }

                    // 添加响应函数
                    dependList.add(activeEffect);
                    return target[key];
                },
                set(target, key, newVal) {
                    target[key] = newVal;
                    const dependMap = bucket.get(target);
                    // 该对象没有设置响应关系
                    if (!dependMap) return;
                    const dependList = dependMap.get(key);
                    // dependList存在则遍历执行
                    dependList && dependList.forEach((f) => f());
                },
            });
            // 数据修改1
            function addData() {
                if (objP.index >= 10) {
                    return window.alert("到顶啦");
                }
                objP.index += 1;
            }
            // 数据修改2
            function curData() {
                if (objP.index <= 1) {
                    return window.alert("到底啦");
                }
                objP.index -= 1;
            }

            // 注册响应函数
            effect(() => {
                document.getElementById("wrapper").innerText = `${data.index}`;
            });
```

最后将get和set的代码进行封装

```html
<body>
        <span id="wrapper"></span>
        <button id="btn" onclick="addData()">+1</button>
        <button id="btn" onclick="curData()">-1</button>
        <div></div>
        <script>
            const data = { index: 1 };
            // “桶”改造
            const bucket = new WeakMap();

            // 改造1 - 响应函数
            let activeEffect;
            function effect(fn) {
                activeEffect = fn;
                fn();
            }
            // 追踪
            function track(target, key) {
                if (!activeEffect) return;
                // 获取响应对象内部的响应关系
                let dependMap = bucket.get(target);
                if (!dependMap) {
                    bucket.set(target, new Map());
                    dependMap = bucket.get(target);
                }
                // 获取对应属性的响应函数列表
                // 存储着与key有关联的响应函数
                let dependList = dependMap[key];
                if (!dependList) {
                    dependMap.set(key, new Set());
                    dependList = dependMap.get(key);
                }

                // 添加响应函数
                dependList.add(activeEffect);
            }

            function trigger(target, key) {
                const dependMap = bucket.get(target);
                // 该对象没有设置响应关系
                if (!dependMap) return;
                const dependList = dependMap.get(key);
                // dependList存在则遍历执行
                dependList && dependList.forEach((f) => f());
            }
            // 改造1 - 拦截代理
            const objP = new Proxy(data, {
                get(target, key) {
                    // 将副作用函数添加进关系中
                    track(target, key);
                    return target[key];
                },
                set(target, key, newVal) {
                    target[key] = newVal;
                    trigger(target, key);
                },
            });
            // 数据修改1
            function addData() {
                if (objP.index >= 10) {
                    return window.alert("到顶啦");
                }
                objP.index += 1;
            }
            // 数据修改2
            function curData() {
                if (objP.index <= 1) {
                    return window.alert("到底啦");
              
              }
                objP.index -= 1;
            }

            // 注册响应函数
            effect(() => {
                document.getElementById("wrapper").innerText = false ? `${objP.index}` : '哈哈哈';
            });
        </script>
    </body>
```
